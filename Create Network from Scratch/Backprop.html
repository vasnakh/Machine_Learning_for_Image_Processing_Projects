def normalize_MNIST_images(x):
    min_p = -1.0
    max_p = 1.0
    min_x = np.min(x)
    max_x = np.max(x)
    x = x.astype(np.float64)
    x = (max_p - min_p)/(max_x-min_x)*(x-max_x)+max_p
    return x

def softmax(a):
    M_all_data = a.max(axis=0)
    a_minus_M = a - M_all_data
    exp_a_minus_M = np.exp(a_minus_M)
    denominator = exp_a_minus_M.sum(axis=0)
    denom_brd_cst = np.tile(denominator, [10,1])
    y = np.divide(exp_a_minus_M, denom_brd_cst)
    return y

def softmaxp(a,e):
    g_a = softmax(a)
    dot_prod = np.zeros((1,a.shape[1]))
    for i in range(0, a.shape[1]):
        dot_prod[0,i] = g_a[:,i].dot(e[:,i].T)

    delta = g_a*e - dot_prod*g_a
    return delta

def eval_loss(y, d):
    
    E_to_sum = d*np.log(y) + (1-d)*np.log(1-y)
    E = - E_to_sum.sum(axis=1)
    loss = np.sum(E) / (y.shape[0]*y.shape[1])
    return loss

def update_shallow(x, d, net, gamma=.05):
    W1 = net[0]
    b1 = net[1]
    W2 = net[2]
    b2 = net[3]
    Ni = W1.shape[1]
    Nh = W1.shape[0]
    No = W2.shape[0]
    gamma = gamma / float(x.shape[1]) # normalized by the training dataset size
    # Forward phase
    a1 = W1.dot(x) + b1
    h1 = relu(a1)
    a2 = W2.dot(h1) + b2
    y = forwardprop_shallow(x, net)
    y_label = onehot2label(y)
    # Error evaluation
    e = -d / y + (1-d) / (1-y)
    # Backward phase
    delta2 = softmaxp(a2,e)
    delta1 = relup(a1, W2.T.dot(delta2)+b2.T.dot(delta2))
    # Gradient update
    W2 = W2 - gamma * delta2.dot(h1.T)
    W1 = W1 - gamma * delta1.dot(x.T)
    b2 = b2 - gamma * delta2.sum(axis=1).reshape(10, 1)
    b1 = b1 - gamma * delta1.sum(axis=1).reshape(64, 1)
    return W1, b1, W2, b2
